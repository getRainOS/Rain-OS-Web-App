public function normalize_content($content_id, $text = '', $html = '', $canonical_url = '') {
  $body = array();
  if (!empty($text)) $body['text'] = $text;
  if (!empty($html)) $body['html'] = $html;
  if (!empty($canonical_url)) $body['canonicalUrl'] = $canonical_url;
  return $this->make_request('api/plugin/content/' . rawurlencode($content_id) . '/normalize', 'POST', $body);
}

public function get_content_analysis($content_id) {
  return $this->make_request('api/plugin/content/' . rawurlencode($content_id) . '/analysis', 'GET');
}

public function get_content_scores($content_id) {
